local FPSCounter = {
    FPS = 0,
    FrameCount = 0,
    LastTime = tick(),
    Display = nil
}

function FPSCounter:Update()
    self.FrameCount = self.FrameCount + 1
    local now = tick()
    local elapsed = now - self.LastTime
    
    if elapsed >= 0.5 then
        self.FPS = math.round(self.FrameCount / elapsed)
        self.FrameCount = 0
        self.LastTime = now
        
        if self.Display then
            self.Display.Text = "📊 FPS: " .. self.FPS
            if self.FPS >= 50 then
                self.Display.TextColor3 = Color3.fromRGB(0, 255, 100)
            elseif self.FPS >= 25 then
                self.Display.TextColor3 = Color3.fromRGB(255, 255, 0)
            else
                self.Display.TextColor3 = Color3.fromRGB(255, 80, 80)
            end
        end
    end
end

-- ============================================
-- OBJECT CACHE SYSTEM
-- ============================================
local ObjectCache = {
    Particles = {},
    Trails = {},
    Fires = {},
    Decals = {},
    BaseParts = {},
    Meshes = {},
    Beams = {},
    Scanned = false
}

local function scanAndCacheObjects()
    if ObjectCache.Scanned then return end
    
    ObjectCache.Particles = {}
    ObjectCache.Trails = {}
    ObjectCache.Fires = {}
    ObjectCache.Decals = {}
    ObjectCache.BaseParts = {}
    ObjectCache.Meshes = {}
    ObjectCache.Beams = {}
    
    local objects = Workspace:GetDescendants()
    
    for _, obj in ipairs(objects) do
        if not obj.Parent then continue end
        
        if obj:IsA("ParticleEmitter") then
            table.insert(ObjectCache.Particles, obj)
        elseif obj:IsA("Trail") then
            table.insert(ObjectCache.Trails, obj)
        elseif obj:IsA("Fire") or obj:IsA("Smoke") then
            table.insert(ObjectCache.Fires, obj)
        elseif obj:IsA("Decal") then
            table.insert(ObjectCache.Decals, obj)
        elseif obj:IsA("BasePart") then
            table.insert(ObjectCache.BaseParts, obj)
        elseif obj:IsA("SpecialMesh") then
            table.insert(ObjectCache.Meshes, obj)
        elseif obj:IsA("Beam") then
            table.insert(ObjectCache.Beams, obj)
        end
    end
    
    ObjectCache.Scanned = true
end

local function applyCachedOptimizations()
    -- Particles
    for _, obj in ipairs(ObjectCache.Particles) do
        if obj and obj.Parent then
            pcall(function() obj.Rate = 0; obj.Enabled = false end)
        end
    end
    
    -- Trails + Fires
    for _, obj in ipairs(ObjectCache.Trails) do
        if obj and obj.Parent then pcall(function() obj.Enabled = false end) end
    end
    for _, obj in ipairs(ObjectCache.Fires) do
        if obj and obj.Parent then pcall(function() obj.Enabled = false end) end
    end
    
    -- Decals
    for _, obj in ipairs(ObjectCache.Decals) do
        if obj and obj.Parent then pcall(function() obj.Transparency = 1 end) end
    end
    
    -- BaseParts (task.defer batch)
    local partIndex = 1
    local function processParts()
        local count = 0
        while partIndex <= #ObjectCache.BaseParts and count < 200 do
            local obj = ObjectCache.BaseParts[partIndex]
            if obj and obj.Parent then
                pcall(function()
                    obj.CastShadow = false
                    obj.Reflectance = 0
                    if obj.Transparency > 0.5 then obj.Material = Enum.Material.SmoothPlastic end
                end)
            end
            partIndex = partIndex + 1
            count = count + 1
        end
        if partIndex <= #ObjectCache.BaseParts then
            task.defer(processParts)
        end
    end
    task.defer(processParts)
    
    -- Meshes
    for _, obj in ipairs(ObjectCache.Meshes) do
        if obj and obj.Parent then pcall(function() obj.TextureId = "" end) end
    end
    
    -- Beams
    for _, obj in ipairs(ObjectCache.Beams) do
        if obj and obj.Parent then pcall(function() obj.Enabled = false end) end
    end
end

-- ============================================
-- FPS BOOSTER
-- ============================================
local function BoostFPS()
    -- Phase 1: Lighting
    pcall(function()
        Lighting.GlobalShadows = false
        Lighting.FogEnd = 999999
        Lighting.FogStart = 999999
        Lighting.Brightness = 3
        Lighting.OutdoorAmbient = Color3.fromRGB(200, 200, 200)
        Lighting.Ambient = Color3.fromRGB(180, 180, 180)
        Lighting.ClockTime = 14
        Lighting.Technology = Enum.Technology.Compatibility
        
        local effects = {"Atmosphere", "Bloom", "Blur", "ColorCorrection", "DepthOfField", "SunRays", "Sky"}
        for _, name in ipairs(effects) do
            local effect = Lighting:FindFirstChild(name)
            if effect then effect.Enabled = false end
        end
    end)
    
    -- Phase 2: Terrain
    pcall(function()
        local terrain = Workspace:FindFirstChild("Terrain")
        if terrain then
            terrain.WaterWaveSize = 0
            terrain.WaterWaveSpeed = 0
            terrain.WaterReflectance = 0
            terrain.WaterTransparency = 0.9
        end
    end)
    
    -- Phase 3: Scan + Apply cache
    scanAndCacheObjects()
    applyCachedOptimizations()
end

UserInputService.InputChanged:Connect(function(input)
    if not dragging or not Main.Visible then return end
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        local delta = input.Position - dragStart
        local camera = Workspace.CurrentCamera
        if not camera then return end
        local screenSize = camera.ViewportSize
        local guiSize = Main.AbsoluteSize
        local inset = GuiService:GetGuiInset()
        local newX = math.clamp(startPos.X.Offset + delta.X, 0, screenSize.X - guiSize.X)
        local newY = math.clamp(startPos.Y.Offset + delta.Y, inset.Y, screenSize.Y - guiSize.Y - inset.Y)
        Main.Position = UDim2.new(0, newX, 0, newY)
    end
end)

    btn.Size = UDim2.new(1, 0, 1, 0)
    btn.BackgroundTransparency = 1
    btn.Text = emoji .. "  " .. name
    btn.TextColor3 = Color3.fromRGB(220, 220, 220)

    btn.TextSize = 14
    btn.ZIndex = 7
    btn.AutoButtonColor = false
    btn.Active = true
    btn.Selectable = true
    btn.Parent = btnFrame
    
    btn.MouseEnter:Connect(function()
        TweenService:Create(stroke, TweenInfo.new(0.15), {Transparency = 0, Thickness = 2.5}):Play()
        TweenService:Create(glow, TweenInfo.new(0.15), {BackgroundTransparency = 0.7}):Play()
        TweenService:Create(btn, TweenInfo.new(0.15), {TextColor3 = Color3.fromRGB(255, 255, 255)}):Play()
    end)

        TweenService:Create(stroke, TweenInfo.new(0.2), {Transparency = 0.4, Thickness = 1.5}):Play()

        end
    end)
    
    return btn, glow
end
