loadstring(game:HttpGet('https://raw.githubusercontent.com/GhostPlayer352/UI-Library/refs/heads/main/Ghost%20Gui'))()
game.CoreGui.GhostGui.MainFrame.Title.Text = "Silent Aim - Made by Khang"

-- // === CHỐNG CHỈNH SỬA (PROTECT) ===
local originalScript = string.dump(function() end) -- Lưu bản sao
local scriptHash = 0
for i = 1, #script.Source do
    scriptHash = scriptHash + string.byte(script.Source, i)
end

local function antiTamper()
    local currentHash = 0
    for i = 1, #script.Source do
        currentHash = currentHash + string.byte(script.Source, i)
    end
    if currentHash ~= scriptHash then
        print("[PROTECT] Phát hiện chỉnh sửa mã nguồn! Tự động khôi phục...")
        script.Source = originalScript
        -- Reset toàn bộ
        enabled = false
        targetPlayer = nil
        mode = "free"
    end
end

-- Kiểm tra mỗi 5 giây
game:GetService("RunService").Heartbeat:Connect(function()
    antiTamper()
end)

-- // === CẤU HÌNH ===
local player = game.Players.LocalPlayer
local mouse = player:GetMouse()
local camera = game.Workspace.CurrentCamera
local runService = game:GetService("RunService")
local enabled = false
local mode = "free" -- free hoặc target
local targetPlayer = nil
local silentAimAngle = 15
local silentAimSmoothness = 0.3
local isMobile = game:GetService("UserInputService").TouchEnabled

-- // === HÀM LẤY PLAYER GẦN NHẤT ===
local function getNearestPlayer()
    local closest = nil
    local minDist = math.huge
    for _, plr in pairs(game.Players:GetPlayers()) do
        if plr ~= player and plr.Character and plr.Character:FindFirstChild("Head") then
            local headPos = plr.Character.Head.Position
            local screenPos, onScreen = camera:WorldToScreenPoint(headPos)
            if onScreen then
                local dist = (Vector2.new(mouse.X, mouse.Y) - Vector2.new(screenPos.X, screenPos.Y)).Magnitude
                if dist < minDist then
                    minDist = dist
                    closest = plr
                end
            end
        end
    end
    return closest
end

-- // === SILENT AIM XỬ LÝ ===
local function silentAim()
    if not enabled then return end
    if not mouse.Button1Down then return end
    
    local target = nil
    if mode == "target" then
        target = targetPlayer
    else
        target = getNearestPlayer()
    end
    
    if target and target.Character and target.Character:FindFirstChild("Head") then
        local headPos = target.Character.Head.Position
        local angle = math.deg(math.acos(camera.CFrame.LookVector:Dot((headPos - camera.CFrame.Position).Unit)))
        
        if angle <= silentAimAngle then
            local newCFrame = CFrame.new(camera.CFrame.Position, headPos)
            camera.CFrame = camera.CFrame:Lerp(newCFrame, silentAimSmoothness)
        end
    end
end

-- // === TẠO UI ===
-- // TextButton: CHỌN NGƯỜI
AddContent("TextButton", "Chọn Target", [[
    local input = game:GetService("GuiService"):GetText()
    if input and input ~= "" then
        for _, p in pairs(game.Players:GetPlayers()) do
            if p.Name:lower():find(input:lower()) then
                targetPlayer = p
                print("[KHANG] Target: " .. p.Name)
                break
            end
        end
    end
]])

-- // Toggle: BẬT/TẮT
AddContent("Toggle", "Bật Silent Aim", [[
    enabled = not enabled
    print(enabled and "[KHANG] Silent Aim: BẬT" or "[KHANG] Silent Aim: TẮT")
]])

-- // Switch: CHẾ ĐỘ
AddContent("Switch", "Chế Độ", [[
    mode = "free"
    print("[KHANG] Chế độ: TỰ DO")
]], [[
    mode = "target"
    print("[KHANG] Chế độ: CHỌN NGƯỜI")
]])

-- // TextBox: TARGET
local targetBox = AddContent("TextBox", "Tên Target")
targetBox:GetPropertyChangedSignal("Text"):Connect(function()
    local name = targetBox.Text
    if name and name ~= "" then
        for _, p in pairs(game.Players:GetPlayers()) do
            if p.Name:lower():find(name:lower()) then
                targetPlayer = p
                print("[KHANG] Target set: " .. p.Name)
                break
            end
        end
    end
end)

-- // TextLabel trạng thái
local statusLabel = AddContent("TextLabel")
statusLabel.Text = "Made by Khang | Status: OFF"

local angleLabel = AddContent("TextLabel")
angleLabel.Text = "Góc: 15° | Mượt: 0.3"

-- // === VÒNG LẶP CHÍNH ===
local lastUpdate = 0
runService.RenderStepped:Connect(function(delta)
    antiTamper()
    lastUpdate = lastUpdate + delta
    if lastUpdate >= 0.08 then
        lastUpdate = 0
        silentAim()
    end
    
    if enabled then
        statusLabel.Text = "Made by Khang | BẬT | " .. (mode == "free" and "TỰ DO" or "CHỌN: " .. (targetPlayer and targetPlayer.Name or "???"))

    else
        statusLabel.Text = "Made by Khang | TẮT"
    end
end)

-- // === MOBILE SUPPORT ===
if isMobile then
    game:GetService("UserInputService").TouchEnded:Connect(function()
        if enabled then silentAim() end
    end)
end
